\newcommand{\plasduinodoctext}[1]%
{
  Una volta acceso il calcolatore, selezionare dalla barra delle applicazioni
  (in basso) l'icona con #1. Questo dovrebbe aprire 
  automaticamente una shell e mostrare la finestra principale del programma di acquisizione.
}


\newcommand{\plasduinodoc}[1]%
{
  \labsection{Note sul programma di acquisizione}

  \plasduinodoctext{#1}
}

\newcommand{\plasduinosave}%
{
  Al termine di ogni sessione di presa dati -- dopo aver premuto il tasto~\menuitem{stop} --
  i dati vengono salvati automaticamente in una cartella visibile dalla finestra~\menuitem{Event handler}.
  Ricordatevi si scrivere sul~\emph{logbook} il numero del~\menuitem{Test stand}, 
  quello del~\menuitem{Run ID} e le condizioni della presa dati!
}
